# Write a function to help visualize the image in Euclidean coordinates
def show_image(xy, colors):
    plt.figure(figsize=(6, 6))
    plt.scatter(xy[:,0], xy[:,1], c=colors, s=200, marker='o', cmap='jet', vmin=-0.5, vmax=1.5)
    plt.hlines(0, xmin=-6, xmax=6, colors='black', linestyle=':', zorder=0)
    plt.vlines(0, ymin=-6, ymax=6, colors='black', linestyle=':', zorder=0)
    plt.axis('equal')
    ax = plt.gca()
    ax.set_xlim((-5, 5))
    ax.set_ylim((-5, 5))
    ax.grid(linestyle='dashed')
    ax.set_axisbelow(True)
    plt.show()

Step 3: Try various linear transformations¶

  • The usual matrix-vector multiplication $A \overrightarrow{v}$ would multiply a column vector $\overrightarrow{v}$ by a matrix $A$ on the left-hand side.
  • Each point or vector $(x,y)$ is now in a row of xy. So mathematically we need to transpose xy first (so each vector is in a column) before multiplying by $A$ on the left-hand side.
  • However, since $A \overrightarrow{v} = \overrightarrow{v}^T A^T$, we can perform the same computation by multiplying xy on the right-hand side with $A^T$
In [9]:
# Negate x value i.e. mirror along y-axis
A = np.array([[-1, 0],
              [0, 1]])

show_image(xy @ A.T, colors)
